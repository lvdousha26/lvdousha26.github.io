{{- $id := hash.FNV32a .seed -}} {{- $buildKey := now.Unix -}} {{- $url := "" -}} {{- /* Step 1: Get CDN URL from API */ -}} {{- $picURL := "" -}} {{- /* Primary: idnm.de (备用站) — buildKey ensures a fresh API call each build */ -}} {{- $apiURL := printf "https://idnm.de/api.php?sort=CDNiw233&type=json&_=%d_%d" $id $buildKey -}} {{- $resp := resources.GetRemote $apiURL -}} {{- if $resp -}} {{- $body := $resp.Content -}} {{- if $body -}} {{- $parsed := $body | transform.Unmarshal -}} {{- if and $parsed (isset $parsed "pic") -}} {{- $pics := index $parsed "pic" -}} {{- with index $pics 0 -}} {{- $picURL = . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Fallback: cnmiw.com (原站) if primary failed */ -}} {{- if not $picURL -}} {{- $apiURL := printf "https://cnmiw.com/api.php?sort=CDNiw233&type=json&_=%d_%d" $id $buildKey -}} {{- $resp := resources.GetRemote $apiURL -}} {{- if $resp -}} {{- $body := $resp.Content -}} {{- if $body -}} {{- $parsed := $body | transform.Unmarshal -}} {{- if and $parsed (isset $parsed "pic") -}} {{- $pics := index $parsed "pic" -}} {{- with index $pics 0 -}} {{- $picURL = . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Step 2: Download image and serve locally (different each build) */ -}} {{- if $picURL -}} {{- $img := resources.GetRemote $picURL -}} {{- if $img -}} {{- $ext := path.Ext $picURL -}} {{- if not $ext -}}{{- $ext = ".jpg" -}}{{- end -}} {{- $local := $img | resources.Copy (printf "images/covers/%d_%d%s" $id $buildKey $ext) -}} {{- $url = $local.RelPermalink -}} {{- end -}} {{- end -}} {{- return $url -}}
